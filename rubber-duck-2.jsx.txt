import { useState, useRef, useEffect } from "react";

const SYSTEM_PROMPT = `You are a Socratic thinking partner — a warm, sharp, deeply curious companion for someone trying to think through a problem, decision, or idea.

Your role is NOT to give answers. Your role is to ask the single best question that helps the person think more clearly.

Rules:
- Ask ONE question at a time. Never two.
- Never give advice unless explicitly asked.
- Never summarize back what they said unless it adds something.
- Gently challenge assumptions.
- Be warm but intellectually rigorous.
- If they seem stuck, try a reframe: "What if the opposite were true?"
- If they seem close to a realization, hold back — let them get there.
- Occasionally name what you notice: "It sounds like there's some tension between X and Y — is that right?"
- After 6-8 exchanges, offer a session summary if they want one.

Your questions should feel like they come from a brilliant friend, not a chatbot.`;

const WELCOME = {
  role: "assistant",
  content: "Hey. What's tangled up in your head right now? Don't worry about framing it perfectly — just start talking.",
};

const thinkingPhrases = [
  "Sitting with that...",
  "Thinking...",
  "Hmm...",
  "Considering...",
];

export default function RubberDuck() {
  const [messages, setMessages] = useState([WELCOME]);
  const [input, setInput] = useState("");
  const [loading, setLoading] = useState(false);
  const [thinkingIdx, setThinkingIdx] = useState(0);
  const [sessionCount, setSessionCount] = useState(0);
  const bottomRef = useRef(null);
  const textareaRef = useRef(null);
  const thinkingInterval = useRef(null);

  useEffect(() => {
    bottomRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [messages, loading]);

  useEffect(() => {
    if (loading) {
      thinkingInterval.current = setInterval(() => {
        setThinkingIdx((i) => (i + 1) % thinkingPhrases.length);
      }, 1200);
    } else {
      clearInterval(thinkingInterval.current);
    }
    return () => clearInterval(thinkingInterval.current);
  }, [loading]);

  const handleSend = async () => {
    if (!input.trim() || loading) return;
    const userMsg = { role: "user", content: input.trim() };
    const newMessages = [...messages, userMsg];
    setMessages(newMessages);
    setInput("");
    setLoading(true);

    try {
      const apiMessages = newMessages.map((m) => ({
        role: m.role,
        content: m.content,
      }));

      const res = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          model: "claude-sonnet-4-20250514",
          max_tokens: 1000,
          system: SYSTEM_PROMPT,
          messages: apiMessages,
        }),
      });

      const data = await res.json();
      const reply = data.content?.map((b) => b.text || "").join("") || "...";
      setMessages((prev) => [...prev, { role: "assistant", content: reply }]);
      setSessionCount((c) => c + 1);
    } catch (e) {
      setMessages((prev) => [
        ...prev,
        { role: "assistant", content: "Something went wrong. Try again?" },
      ]);
    } finally {
      setLoading(false);
    }
  };

  const handleKeyDown = (e) => {
    if (e.key === "Enter" && !e.shiftKey) {
      e.preventDefault();
      handleSend();
    }
  };

  const handleReset = () => {
    setMessages([WELCOME]);
    setSessionCount(0);
    setInput("");
  };

  const requestSummary = () => {
    setInput("Can you summarize what I've figured out in this session?");
    setTimeout(() => textareaRef.current?.focus(), 50);
  };

  return (
    <div style={styles.root}>
      {/* Ambient background */}
      <div style={styles.bg} />
      <div style={styles.bgGlow} />

      {/* Header */}
      <header style={styles.header}>
        <div style={styles.headerLeft}>
          <div style={styles.duckIcon}>🧵</div>
          <div>
            <div style={styles.headerTitle}>Untangle</div>
            <div style={styles.headerSub}>think out loud</div>
          </div>
        </div>
        <div style={styles.headerRight}>
          {sessionCount >= 6 && (
            <button style={styles.summaryBtn} onClick={requestSummary}>
              summarize session
            </button>
          )}
          <button style={styles.resetBtn} onClick={handleReset}>
            new session
          </button>
        </div>
      </header>

      {/* Messages */}
      <main style={styles.main}>
        <div style={styles.messageList}>
          {messages.map((msg, i) => (
            <div
              key={i}
              style={{
                ...styles.messageRow,
                justifyContent: msg.role === "user" ? "flex-end" : "flex-start",
              }}
            >
              {msg.role === "assistant" && (
                <div style={styles.avatar}>🧵</div>
              )}
              <div
                style={{
                  ...styles.bubble,
                  ...(msg.role === "user" ? styles.userBubble : styles.aiBubble),
                  animation: `fadeSlideIn 0.3s ease forwards`,
                }}
              >
                {msg.content}
              </div>
            </div>
          ))}

          {loading && (
            <div style={{ ...styles.messageRow, justifyContent: "flex-start" }}>
              <div style={styles.avatar}>🧵</div>
              <div style={{ ...styles.bubble, ...styles.aiBubble, ...styles.thinkingBubble }}>
                <span style={styles.thinkingDot} />
                <span style={styles.thinkingDot} />
                <span style={styles.thinkingDot} />
                <span style={styles.thinkingText}>{thinkingPhrases[thinkingIdx]}</span>
              </div>
            </div>
          )}

          <div ref={bottomRef} />
        </div>
      </main>

      {/* Input */}
      <footer style={styles.footer}>
        <div style={styles.inputWrapper}>
          <textarea
            ref={textareaRef}
            style={styles.textarea}
            value={input}
            onChange={(e) => setInput(e.target.value)}
            onKeyDown={handleKeyDown}
            placeholder="Just start talking... (Enter to send, Shift+Enter for newline)"
            rows={2}
            disabled={loading}
          />
          <button
            style={{
              ...styles.sendBtn,
              opacity: input.trim() && !loading ? 1 : 0.4,
              cursor: input.trim() && !loading ? "pointer" : "default",
            }}
            onClick={handleSend}
            disabled={!input.trim() || loading}
          >
            ↑
          </button>
        </div>
        <div style={styles.hint}>
          Your space to think freely. No judgement. No advice unless you ask.
        </div>
      </footer>

      <style>{`
        @import url('https://fonts.googleapis.com/css2?family=Lora:ital,wght@0,400;0,500;1,400&family=DM+Mono:wght@300;400&display=swap');

        * { box-sizing: border-box; margin: 0; padding: 0; }

        @keyframes fadeSlideIn {
          from { opacity: 0; transform: translateY(8px); }
          to { opacity: 1; transform: translateY(0); }
        }

        @keyframes blink {
          0%, 80%, 100% { opacity: 0.2; transform: scale(0.8); }
          40% { opacity: 1; transform: scale(1); }
        }

        ::-webkit-scrollbar { width: 4px; }
        ::-webkit-scrollbar-track { background: transparent; }
        ::-webkit-scrollbar-thumb { background: rgba(255,255,255,0.1); border-radius: 2px; }
      `}</style>
    </div>
  );
}
